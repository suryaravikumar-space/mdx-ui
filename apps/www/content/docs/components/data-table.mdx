## Installation

```bash
npx @ravikumarsurya/mdx-ui add data-table
```

## Usage

Pass `columns` and `data` as props. Add `searchable` to enable filtering, and `sortable: true` on any column to enable click-to-sort.

```mdx
export const columns = [
  { key: "name",    header: "Component", sortable: true },
  { key: "version", header: "Since",     sortable: true },
  { key: "status",  header: "Status" },
]

export const data = [
  { name: "Accordion", version: "0.0.1",  status: "stable" },
  { name: "DataTable", version: "0.0.27", status: "stable" },
  { name: "Mermaid",   version: "0.0.17", status: "stable" },
]

<DataTable columns={columns} data={data} searchable />
```

## Examples

### Sortable + searchable

export const componentColumns = [
  { key: "name",    header: "Component", sortable: true },
  { key: "version", header: "Since",     sortable: true },
  { key: "status",  header: "Status",    sortable: true },
]

export const componentData = [
  { name: "Accordion",  version: "0.0.1",  status: "stable" },
  { name: "Alert",      version: "0.0.20", status: "stable" },
  { name: "Badge",      version: "0.0.1",  status: "stable" },
  { name: "Callout",    version: "0.0.1",  status: "stable" },
  { name: "Card",       version: "0.0.20", status: "stable" },
  { name: "Changelog",  version: "0.0.26", status: "stable" },
  { name: "DataTable",  version: "0.0.27", status: "stable" },
  { name: "DiffBlock",  version: "0.0.26", status: "stable" },
  { name: "Mermaid",    version: "0.0.17", status: "stable" },
  { name: "Terminal",   version: "0.0.26", status: "stable" },
]

<DataTable columns={componentColumns} data={componentData} searchable searchPlaceholder="Search components…" />

### Custom cell renderer

```mdx
export const columns = [
  { key: "name",   header: "Name", sortable: true },
  { key: "status", header: "Status", render: (value) => (
    <span className={value === "stable" ? "text-green-600 font-medium" : "text-yellow-600"}>
      {String(value)}
    </span>
  )},
]
```

### From imported JSON

```mdx
import data from "./props.json"

export const columns = [
  { key: "prop",        header: "Prop",        sortable: true },
  { key: "type",        header: "Type" },
  { key: "required",    header: "Required",    sortable: true },
  { key: "description", header: "Description", searchable: false },
]

<DataTable columns={columns} data={data} searchable />
```

### Empty state

```mdx
<DataTable columns={columns} data={[]} emptyText="No results found." />
```

## Props

### DataTable

| Prop | Type | Default | Description |
| --- | --- | --- | --- |
| `columns` | `DataTableColumn[]` | — | Column definitions (required) |
| `data` | `T[]` | — | Array of row objects (required) |
| `searchable` | `boolean` | `false` | Show a search input above the table |
| `searchPlaceholder` | `string` | `"Search…"` | Placeholder text for the search input |
| `caption` | `string` | — | Caption rendered below the table |
| `emptyText` | `string` | `"No data available."` | Text shown when data is empty |
| `className` | `string` | — | Extra classes on the outer wrapper |

### DataTableColumn

| Prop | Type | Default | Description |
| --- | --- | --- | --- |
| `key` | `string` | — | Key to read from each row object (required) |
| `header` | `string` | — | Column header label (required) |
| `sortable` | `boolean` | `false` | Click header to sort asc → desc → off |
| `searchable` | `boolean` | `true` | Include column in search filtering |
| `render` | `(value, row, index) => ReactNode` | — | Custom cell renderer |
| `className` | `string` | — | Extra classes on `<th>` and `<td>` |
